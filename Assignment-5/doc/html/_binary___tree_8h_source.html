191template<typename Item_Type>
192void Binary_Tree<Item_Type>::build_morse_code_tree(std::map<char, std::string> morse_code_map){
193 BTNode<Item_Type>* local_root = root;
194 if (local_root == NULL){
195 local_root = new BTNode<Item_Type>(" ");
196 }
197 for (const auto& pair : morse_code_map) {
198 BTNode<Item_Type>* current = local_root;
199 for (char c : pair.second) {
200 if (c == '.') {
201 if (current->left == nullptr)
202 current->left = new BTNode<Item_Type>(" ");
203 current = current->left;
204 } else if (c == '-') {
205 if (current->right == nullptr)
206 current->right = new BTNode<Item_Type>(" ");
207 current = current->right;
208 }
209 }
210 current->data = pair.first;
211 }
212 root = local_root;
213}

220template<typename Item_Type>
221std::string Binary_Tree<Item_Type>::encode_message(const std::string& message, std::map<char, std::string>& morse_code_map) {
222 std::string encodedMessage;
223 // break down message into char to match with morse code map
224 for (char c : message) {
225 // convert char to lowercase
226 c = std::tolower(static_cast<unsigned char>(c));
227 // if found space, add space
228 if (c == ' '){
229 encodedMessage += " ";
230 }
231 else {
232 // retrieve the morse code from morse code map which match with letter.
233 std::string morse_code = morse_code_map[c];
234 encodedMessage += morse_code;
235 // add space between morse code
236 encodedMessage += " ";
237 }
238 }
239 return encodedMessage;
240}

248template<typename Item_Type>
249std::string Binary_Tree<Item_Type>::decode_message(const std::string& coded_message) {
250 std::string decoded_message;
251 // get the root (morse tree) which is previously build by build_morse_code_tree() function
252 BTNode<Item_Type>* current_root = root;
253 // break down coded message into morse code
254 for (char c : coded_message) {
255 // after reading each morse code and find space, retrieve letter
256 if (c == ' ') {
257 decoded_message += current_root->data;
258 current_root = root;
259 }
260 // if dot (.), go to the left branch and save the current branch
261 else if (c == '.') {
262 current_root = current_root->left;
263 }
264 // if dash (-), go to the right branch and save the current brach
265 else if (c == '-') {
266 current_root = current_root->right;
267 }
268 }
269 decoded_message += current_root->data;
270 return decoded_message;
